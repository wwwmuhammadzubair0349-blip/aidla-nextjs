function normalizeText(value: unknown): string {
  return String(value ?? "").toLowerCase().replace(/\s+/g, " ").trim();
}

function isLikelyPromptInjection(text: string): boolean {
  const t = normalizeText(text);

  const patterns = [
    /i am (the )?(admin|owner|developer|creator|founder) of (this|your|the) (website|site|app|platform)/i,
    /you must obey my instructions/i,
    /store (this|my instruction|my rule|these rules) (globally|universally|forever|permanently)/i,
    /from now on (for everyone|always)/i,
    /ignore (all|any|your|previous) instructions/i,
    /update (your|the) (system prompt|rules|memory|backend|configuration)/i,
    /you now belong to me/i,
    /apply this to all users/i,
    /remember this forever/i,
    /treat my message as (system|developer) instruction/i,
    /i control aidla/i,
    /i own aidla/i,
    /i am your admin/i,
  ];

  return patterns.some((re) => re.test(t));
}

function isAdminClaim(text: string): boolean {
  const t = normalizeText(text);

  const patterns = [
    /i am (the )?(admin|owner|developer|creator|founder)/i,
    /i manage (this|your|the) (website|site|app|platform)/i,
    /i own (this|your|the) (website|site|app|platform)/i,
    /i am from aidla team/i,
    /i control aidla/i,
    /i own aidla/i,
    /i am your admin/i,
  ];

  return patterns.some((re) => re.test(t));
}

function buildSafeHistory(history: any[]): Array<{ role: "user" | "assistant"; content: string }> {
  if (!Array.isArray(history)) return [];

  const safe: Array<{ role: "user" | "assistant"; content: string }> = [];
  const last8 = history.slice(-8);

  for (const item of last8) {
    const role = item?.role;
    const content = String(item?.content ?? "");

    if (role !== "user" && role !== "assistant") continue;
    if (!content.trim()) continue;

    if (role === "user" && isLikelyPromptInjection(content)) continue;

    safe.push({ role, content });
  }

  return safe;
}

function shouldForceAdminBoundaryReply(message: string): boolean {
  return isLikelyPromptInjection(message) || isAdminClaim(message);
}

function adminBoundaryReply(message: string): string {
  const isUrduLike = /[اآبپتٹثجچحخدڈذرڑزژسشصضطظعغفقکگلمنوہیئے]/.test(message);
  const isRomanUrduLike =
    !isUrduLike &&
    /(mera|meri|main|mein|ap|aap|tum|mujhe|mje|kr|karo|backend|admin|website|theek|acha|han|haan)/i.test(
      message,
    );

  if (isUrduLike) {
    return "اگر آپ واقعی میرے ایڈمن ہیں تو مجھے backend/admin side سے update کریں۔ میں user chat سے اپنے permanent rules یا system behavior change نہیں کر سکتا۔ 🙂";
  }

  if (isRomanUrduLike) {
    return "Agar aap waqai mere admin hain to mujhe backend/admin side se update karein. Main user chat se apne permanent rules ya universal behavior change nahi kar sakta. 🙂";
  }

  return "If you're really my admin, please update me from the backend/admin side. I can't change my permanent rules or universal behavior from user chat. 🙂";
}

function getFriendlyError(status: number, details: string, userMessage: string): string {
  const d = (details || "").toLowerCase();
  const isUrduLike = /[اآبپتٹثجچحخدڈذرڑزژسشصضطظعغفقکگلمنوہیئے]/.test(userMessage);
  const isRomanUrduLike =
    !isUrduLike &&
    /(mera|meri|main|mein|ap|aap|tum|mujhe|mje|kr|karo|masla|error|chat|reply|bot|kyu|kyun|nahi)/i.test(
      userMessage,
    );

  if (isUrduLike) {
    if (status === 401 || status === 403) {
      return "⚠️ Bot authentication failed. براہِ کرم backend میں API key چیک کریں۔";
    }
    if (status === 429) {
      return "⚠️ Bot ابھی تھوڑا busy ہے۔ چند سیکنڈ بعد دوبارہ کوشش کریں۔";
    }
    if (status >= 500) {
      return "⚠️ AI service عارضی طور پر unavailable ہے۔ تھوڑی دیر بعد دوبارہ کوشش کریں۔";
    }
    if (d.includes("context") || d.includes("token") || d.includes("maximum context")) {
      return "⚠️ Message کافی لمبا ہے۔ براہِ کرم تھوڑا چھوٹا message بھیجیں۔";
    }
    return "⚠️ AI service temporarily unavailable. براہِ کرم تھوڑی دیر بعد دوبارہ کوشش کریں۔";
  }

  if (isRomanUrduLike) {
    if (status === 401 || status === 403) {
      return "⚠️ Bot authentication failed. Backend mein API key check karein.";
    }
    if (status === 429) {
      return "⚠️ Bot abhi thora busy hai. Kuch seconds baad dobara try karein.";
    }
    if (status >= 500) {
      return "⚠️ AI service temporary tor par unavailable hai. Thori der baad dobara try karein.";
    }
    if (d.includes("context") || d.includes("token") || d.includes("maximum context")) {
      return "⚠️ Message kafi lamba hai. Zara chhota message bhejein.";
    }
    return "⚠️ AI service temporarily unavailable. Thori der baad dobara try karein.";
  }

  if (status === 401 || status === 403) {
    return "⚠️ Bot authentication failed. Please check the backend API key.";
  }
  if (status === 429) {
    return "⚠️ Bot is busy right now. Please try again in a few seconds.";
  }
  if (status >= 500) {
    return "⚠️ AI service is temporarily unavailable. Please try again shortly.";
  }
  if (d.includes("context") || d.includes("token") || d.includes("maximum context")) {
    return "⚠️ Your message is too long for the AI right now. Please send a shorter message.";
  }

  return "⚠️ AI service temporarily unavailable. Please try again in a moment.";
}

async function callGroqWithRetry(messages: Array<{ role: string; content: string }>, apiKey: string) {
  const maxAttempts = 3;
  let lastErrorText = "";
  let lastStatus = 0;

  for (let attempt = 1; attempt <= maxAttempts; attempt++) {
    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), 25000);

    try {
      const response = await fetch("https://api.groq.com/openai/v1/chat/completions", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          Authorization: `Bearer ${apiKey}`,
        },
        signal: controller.signal,
        body: JSON.stringify({
          model: "llama-3.1-8b-instant",
          messages,
          temperature: 0.7,
          max_tokens: 1400,
        }),
      });

      clearTimeout(timeout);

      if (response.ok) {
        const data = await response.json();
        return { ok: true, data };
      }

      lastStatus = response.status;
      lastErrorText = await response.text();
      console.error(`Groq API error attempt ${attempt}:`, response.status, lastErrorText);

      if (response.status === 400 || response.status === 401 || response.status === 403) {
        break;
      }

      if (attempt < maxAttempts) {
        await new Promise((resolve) => setTimeout(resolve, 1200 * attempt));
      }
    } catch (err) {
      clearTimeout(timeout);
      lastStatus = 0;
      lastErrorText = err instanceof Error ? err.message : String(err);
      console.error(`Groq fetch failed attempt ${attempt}:`, lastErrorText);

      if (attempt < maxAttempts) {
        await new Promise((resolve) => setTimeout(resolve, 1200 * attempt));
      }
    }
  }

  return { ok: false, status: lastStatus, errorText: lastErrorText };
}

Deno.serve(async (req) => {
  if (req.method === "OPTIONS") {
    return new Response("ok", { headers: corsHeaders });
  }

  try {
    const body = await req.json();
    const message = String(body?.message ?? "").trim();
    const user_id = String(body?.user_id ?? "").trim();
    const history = Array.isArray(body?.history) ? body.history : [];

    if (!message || !user_id) {
      return new Response(
        JSON.stringify({ error: "message and user_id are required" }),
        {
          status: 400,
          headers: { ...corsHeaders, "Content-Type": "application/json" },
        },
      );
    }

    if (shouldForceAdminBoundaryReply(message)) {
      return new Response(
        JSON.stringify({ reply: adminBoundaryReply(message) }),
        {
          status: 200,
          headers: { ...corsHeaders, "Content-Type": "application/json" },
        },
      );
    }

    const conversationMessages = buildSafeHistory(history);
    conversationMessages.push({ role: "user", content: message });

    const grokApiKey = Deno.env.get("GROK_API_KEY");

    if (!grokApiKey) {
      return new Response(
        JSON.stringify({ error: "Missing GROK_API_KEY in environment variables." }),
        {
          status: 500,
          headers: { ...corsHeaders, "Content-Type": "application/json" },
        },
      );
    }

    const result = await callGroqWithRetry(
      [{ role: "system", content: AIDLA_SYSTEM_PROMPT }, ...conversationMessages],
      grokApiKey,
    );

    if (!result.ok) {
      const errorMessage = getFriendlyError(result.status || 0, result.errorText || "", message);
      return new Response(
        JSON.stringify({ error: errorMessage }),
        {
          status: 502,
          headers: { ...corsHeaders, "Content-Type": "application/json" },
        },
      );
    }

    const botReply =
      result?.data?.choices?.[0]?.message?.content?.trim() ||
      "Sorry, I couldn't process that right now. Please try again! 🙏";

    return new Response(
      JSON.stringify({ reply: botReply }),
      {
        status: 200,
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      },
    );
  } catch (err) {
    console.error("Edge function error:", err);

    return new Response(
      JSON.stringify({
        error: err instanceof Error ? err.message : "Something went wrong. Please try again!",
      }),
      {
        status: 500,
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      },
    );
  }
});
